# --- Sections ---

def get_sections(grade_level_id=None, db_path=None):
    conn = get_db(db_path)
    query = (
        "SELECT s.id, s.grade_level_id, g.name as grade_name, s.section_number, "
        "s.homeroom_teacher_id, t.name as teacher_name, t.color as teacher_color, "
        "s.classroom_id, c.name as classroom_name "
        "FROM sections s "
        "JOIN grade_levels g ON s.grade_level_id = g.id "
        "LEFT JOIN teachers t ON s.homeroom_teacher_id = t.id "
        "LEFT JOIN classrooms c ON s.classroom_id = c.id "
    )
    if grade_level_id:
        query += "WHERE s.grade_level_id = ? "
        query += "ORDER BY g.name, s.section_number"
        rows = conn.execute(query, (grade_level_id,)).fetchall()
    else:
        query += "ORDER BY g.name, s.section_number"
        rows = conn.execute(query).fetchall()
    conn.close()
    return [dict(r) for r in rows]
```

def set_section_count(grade_level_id, count, db_path=None):
    conn = get_db(db_path)
    current = conn.execute(
        "SELECT COUNT(*) FROM sections WHERE grade_level_id = ?",
        (grade_level_id,)
    ).fetchone()[0]

    if count > current:
        for sec_num in range(current + 1, count + 1):
            conn.execute(
                "INSERT INTO sections (grade_level_id, section_number) VALUES (?, ?)",
                (grade_level_id, sec_num)
            )
    elif count < current:
        conn.execute(
            "DELETE FROM sections WHERE grade_level_id = ? AND section_number > ?",
            (grade_level_id, count)
        )

    conn.commit()
    conn.close()
```

def assign_homeroom_teacher(section_id, teacher_id, db_path=None):
    conn = get_db(db_path)
    if teacher_id is None:
        conn.execute(
            "UPDATE sections SET homeroom_teacher_id = NULL WHERE id = ?",
            (section_id,)
        )
        conn.commit()
        conn.close()
        return True, None

    # Clear any existing section assignment for this teacher
    conn.execute(
        "UPDATE sections SET homeroom_teacher_id = NULL WHERE homeroom_teacher_id = ?",
        (teacher_id,)
    )
    conn.execute(
        "UPDATE sections SET homeroom_teacher_id = ? WHERE id = ?",
        (teacher_id, section_id)
    )
    conn.commit()
    conn.close()
    return True, None
```

# --- Sections ---

@app.route('/api/sections', methods=['GET'])
def api_get_sections():
    grade_level_id = request.args.get('grade_level_id', type=int)
    return jsonify(get_sections(grade_level_id))


@app.route('/api/gradelevels/<int:grade_level_id>/sections', methods=['PUT'])
def api_set_section_count(grade_level_id):
    data = request.get_json()
    count = data.get('count')
    if count is None or count < 0:
        return jsonify({'error': 'count must be a non-negative integer'}), 400
    set_section_count(grade_level_id, count)
    return jsonify({'message': f'Section count set to {count}.'})


@app.route('/api/sections/<int:section_id>/teacher', methods=['PUT'])
def api_assign_homeroom_teacher(section_id):
    data = request.get_json()
    teacher_id = data.get('teacher_id')  # None to clear
    ok, error = assign_homeroom_teacher(section_id, teacher_id)
    if error:
        return jsonify({'error': error}), 409
    return jsonify({'message': 'Homeroom teacher updated.'})


@app.route('/api/sections/<int:section_id>/classroom', methods=['PUT'])
def api_assign_section_classroom(section_id):
    data = request.get_json()
    classroom_id = data.get('classroom_id')  # None to clear
    ok, error = assign_section_classroom(section_id, classroom_id)
    if error:
        return jsonify({'error': error}), 409
    return jsonify({'message': 'Section classroom updated.'})

  // --- Grades & Sections Page ---
  function renderGradesSectionsPage() {
    const container = document.getElementById('grades-sections-list');
    if (!container) return;

    if (!gradeLevels.length) {
      container.innerHTML = '<div class="empty-state"><p>No grade levels defined.</p></div>';
      return;
    }

    // Find which teacher IDs and classroom IDs are already assigned as homeroom
    const assignedTeacherIds = new Set(sections.filter(s => s.homeroom_teacher_id).map(s => s.homeroom_teacher_id));
    const assignedClassroomIds = new Set(sections.filter(s => s.classroom_id).map(s => s.classroom_id));

    container.innerHTML = gradeLevels.map(grade => {
      const gradeSections = sections.filter(s => s.grade_level_id === grade.id);
      const currentCount = gradeSections.length;

      let tableHtml = '';
      if (gradeSections.length > 0) {
        tableHtml = `
          <table class="section-table">
            <thead>
              <tr>
                <th style="width:50px;">#</th>
                <th>Homeroom Teacher</th>
                <th>Classroom</th>
              </tr>
            </thead>
            <tbody>
              ${gradeSections.map(sec => {
                const teacherOptions = '<option value="">— Select —</option>' +
                  teachers.map(t => {
                    const taken = assignedTeacherIds.has(t.id) && t.id !== sec.homeroom_teacher_id;
                    return `<option value="${t.id}" ${t.id === sec.homeroom_teacher_id ? 'selected' : ''} ${taken ? 'disabled' : ''}>${t.name}${taken ? ' (assigned)' : ''}</option>`;
                  }).join('');

                const classroomOptions = '<option value="">— Select —</option>' +
                  classrooms.map(c => {
                    const taken = assignedClassroomIds.has(c.id) && c.id !== sec.classroom_id;
                    return `<option value="${c.id}" ${c.id === sec.classroom_id ? 'selected' : ''} ${taken ? 'disabled' : ''}>${c.name}${taken ? ' (assigned)' : ''}</option>`;
                  }).join('');

                return `
                  <tr>
                    <td style="font-weight:600;color:var(--text-secondary);">Section ${sec.section_number}</td>
                    <td><select onchange="assignHomeroomTeacher(${sec.id}, this.value)">${teacherOptions}</select></td>
                    <td><select onchange="assignSectionClassroom(${sec.id}, this.value)">${classroomOptions}</select></td>
                  </tr>`;
              }).join('')}
            </tbody>
          </table>`;
      }

      return `
        <div class="grade-card">
          <div class="grade-card-header">
            <h4>${grade.name}</h4>
            <label style="font-size:13px;color:var(--text-secondary);">Sections:</label>
            <input type="number" min="0" max="20" value="${currentCount}" id="section-count-${grade.id}">
            <button class="btn btn-sm" onclick="updateSectionCount(${grade.id})">Update</button>
          </div>
          ${tableHtml}
        </div>`;
    }).join('');
  }
```

  async function updateSectionCount(gradeId) {
    const input = document.getElementById('section-count-' + gradeId);
    const count = parseInt(input.value);
    if (isNaN(count) || count < 0) {
      showToast('Invalid section count.', true);
      return;
    }
    const res = await fetch(`/api/gradelevels/${gradeId}/sections`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ count })
    });
    const data = await res.json();
    if (data.error) {
      showToast(data.error, true);
    } else {
      await fetchSections();
      showToast(data.message);
    }
  }

  async function assignHomeroomTeacher(sectionId, value) {
    const teacherId = value ? parseInt(value) : null;
    const res = await fetch(`/api/sections/${sectionId}/teacher`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ teacher_id: teacherId })
    });
    const data = await res.json();
    if (data.error) {
      showToast(data.error, true);
    } else {
      await fetchSections();
      showToast(data.message);
    }
